# 静默播放
`--no-video --really-quiet`

# 设置播放列表循环
`loop-playlist=inf`

# 开启播放列表随机播放（洗牌）
`shuffle`

# (可选) 保存当前的播放列表顺序，下次启动时恢复
# write-filename-in-watch-later-config=yes
# save-position-on-quit=yes

# (可选) 当播放列表中的文件播放失败时，自动跳到下一个
# load-unsafe-playlists=yes

# (可选) 禁用软字幕选择，避免字幕选项干扰连续播放
# sid=no

# (可选) 禁止在播放开始时暂停，确保流畅的随机播放体验
# no-pause
